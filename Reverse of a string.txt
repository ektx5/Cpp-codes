#include <iostream>
#include <cstring>
using namespace std;

int main()
{
    char str[100];
    cout<<"enter a string: ";
    cin>>str;
    int len= strlen(str);
    
    char *ptr=str+len-1;
    cout<<"reversed string is: ";
    while(ptr>=str){
        cout<<*ptr;
        ptr--;
    }

    return 0;
}
